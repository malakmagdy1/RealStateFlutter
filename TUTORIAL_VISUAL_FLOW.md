## Implementation Example

### Screen Structure:
```dart
class _HomeScreenState extends State<HomeScreen> {
  // Step 1: Create GlobalKeys for elements you want to highlight
  final GlobalKey _searchKey = GlobalKey();
  final GlobalKey _filterKey = GlobalKey();
  final GlobalKey _companyKey = GlobalKey();
  final GlobalKey _compoundKey = GlobalKey();

  @override
  void initState() {
    super.initState();
    // Step 2: Show tutorial after UI is built
    WidgetsBinding.instance.addPostFrameCallback((_) {
      _showTutorialIfNeeded();
    });
  }

  // Step 3: Call the tutorial service
  Future<void> _showTutorialIfNeeded() async {
    final tutorialService = TutorialCoachService();
    await Future.delayed(Duration(milliseconds: 500));

    if (mounted) {
      await tutorialService.showHomeTutorial(
        context: context,
        searchKey: _searchKey,    // Pass all keys
        filterKey: _filterKey,
        companyKey: _companyKey,
        compoundKey: _compoundKey,
      );
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Column(
        children: [
          // Step 4: Add keys to widgets
          TextField(
            key: _searchKey, // â† Key here
            decoration: InputDecoration(hintText: 'Search'),
          ),
          IconButton(
            key: _filterKey, // â† Key here
            icon: Icon(Icons.filter_list),
            onPressed: () {},
          ),
          Container(
            key: _companyKey, // â† Key here
            child: Text('Companies'),
          ),
          Container(
            key: _compoundKey, // â† Key here
            child: Text('Compounds'),
          ),
        ],
      ),
    );
  }
}
```
